                // Variabele voor de motoren aan te sturen
                const int stby = 9;
                const int ain1 = 2;
                const int ain2 = 4;
                const int bin1 = 7;
                const int bin2 = 8;
                const int pwma = 5;
                const int pwmb = 6;
                
                // Variabele joystick
                const int vry = A0;
                const int vrx = A1;
                const int sw = 12;
                
                int x = 0;
                int mapx = 0;
                int y = 0;
                int mapy = 0;
                int sw_status = 0;
                
                // Variavele voor Ultrasone
                const int trigPin = 10;
                const int echoPin = 11;
                
                float tijdUltrasone;
                float afstandUltrasone = 0;
                
                int State = LOW;
                unsigned long previousMillis = 0;
                const long interval = 1000;
                
                void setup() {
                
                    Serial.begin(9600);
                    // motoren
                    digitalWrite(stby, HIGH);
                
                    // Joystick
                    pinMode(vrx, INPUT);
                    pinMode(vry, INPUT);
                    pinMode(sw, INPUT_PULLUP);
                
                    // Joystick zekerheid
                    digitalWrite(ain1, LOW);
                    digitalWrite(ain2, LOW);
                    digitalWrite(bin1, LOW);
                    digitalWrite(bin2, LOW);
                
                    x = 0;
                    mapx = 0;
                    y = 0;
                    mapy = 0;
                    sw_status = 0;
                
                    pinMode(trigPin, OUTPUT);
                    pinMode(echoPin, INPUT);
                }
                
                void loop() {
                
                    x = analogRead(vrx);
                    y = analogRead(vry);
                    sw_status = digitalRead(sw);
                
                    mapx = map(x, 0, 1018, -512, 512);
                    mapy = map(y, 0, 1018, -512, 512);
                
                    if (mapx > 500) {
                        digitalWrite(ain1, HIGH);
                        digitalWrite(bin1, HIGH);
                        analogWrite(pwma, 200);
                        analogWrite(pwmb, 200);
                        delay(5);
                        digitalWrite(ain1, LOW);
                        digitalWrite(bin1, LOW);
                        analogWrite(pwma, 0);
                        analogWrite(pwmb, 0);
                    }
                
                    if (mapx < -500) {
                        digitalWrite(ain2, HIGH);
                        digitalWrite(bin2, HIGH);
                        analogWrite(pwma, 200);
                        analogWrite(pwmb, 200);
                        delay(5);
                        digitalWrite(ain2, LOW);
                        digitalWrite(bin2, LOW);
                        analogWrite(pwma, 0);
                        analogWrite(pwmb, 0);
                    }
                
                    if (mapy > 500) {
                        digitalWrite(bin1, HIGH);
                        analogWrite(pwmb, 200);
                        delay(5);
                        digitalWrite(bin1, LOW);
                        analogWrite(pwmb, 0);
                    }
                
                    if (mapy < -500) {
                        digitalWrite(ain1, HIGH);
                        analogWrite(pwma, 200);
                        delay(5);
                        digitalWrite(ain1, LOW);
                        analogWrite(pwma, 0);
                    }
                }
