                // Variabele voor de motoren aan te sturen
                const int stby = 9;
                const int ain1 = 2;
                const int ain2 = 4;
                const int bin1 = 7;
                const int bin2 = 8;
                const int pwma = 5;
                const int pwmb = 6;
                // Variabele voor de 90° te draaien
                const int draai90 = 115; 
                int tijdvoorachter = 1000;
                
                // Variabele voor knop
                const int knop = 13;
                int knopstatus = 0;
                
                // Variabele voor de ultrasone
                const int trigPin = 10;
                const int echoPin = 11;
                
                long tijdUltrasone;
                int afstandUltrasone = 0;
                
                // Variabele joystick
                const int vry = A0;
                const int vrx = A1;
                const int sw = 12;
                
                int x = 0;
                int mapx = 0;
                int y = 0;
                int mapy = 0;
                int sw_status = 0;
                
                
                void setup() {
                
                  // motoren
                  digitalWrite(stby, HIGH);
                  pinMode( knop, INPUT);
                
                  // ultrasone
                  pinMode(trigPin, OUTPUT);
                  pinMode(echoPin, INPUT);
                
                  // Joystick
                  pinMode(vrx, INPUT);
                  pinMode(vry, INPUT);
                  pinMode(sw, INPUT_PULLUP);
                }
                
                void loop() {
                
                  // Knopstatus lezen
                  knopstatus = digitalRead(knop);
                
                  if (knopstatus == HIGH) {
                    delay(500);
                
                    // Ultrasone check
                    digitalWrite(trigPin, LOW);
                    delayMicroseconds(2);
                    digitalWrite(trigPin, HIGH);
                    delayMicroseconds(10);
                    digitalWrite(trigPin, LOW);
                
                    tijdUltrasone = pulseIn(echoPin, HIGH);
                    afstandUltrasone = tijdUltrasone * 0.034 / 2;
                
                    // 3s vooruit
                    digitalWrite(ain1, HIGH);
                    digitalWrite(bin1, HIGH);
                
                    if (afstandUltrasone > 7) {
                      analogWrite(pwma, 255);
                      analogWrite(pwmb, 255);
                      delay(tijdvoorachter);
                      digitalWrite(ain1, LOW);
                      digitalWrite(bin1, LOW);
                      analogWrite(pwma, 0);
                      analogWrite(pwmb, 0);
                    }
                    else {
                      // 180° draaien
                      digitalWrite(ain1, HIGH);
                      analogWrite(pwma, draai90);
                      delay(tijdvoorachter * 2);
                      digitalWrite(ain1, LOW);
                      analogWrite(pwma, 0);
                
                      // Doorgaan
                      analogWrite(pwma, 255);
                      analogWrite(pwmb, 255);
                      delay(tijdvoorachter);
                      digitalWrite(ain1, LOW);
                      digitalWrite(bin1, LOW);
                      analogWrite(pwma, 0);
                      analogWrite(pwmb, 0);
                    }
                
                    delay(1000);
                
                    // Ultrasone check
                    digitalWrite(trigPin, LOW);
                    delayMicroseconds(2);
                    digitalWrite(trigPin, HIGH);
                    delayMicroseconds(10);
                    digitalWrite(trigPin, LOW);
                
                    tijdUltrasone = pulseIn(echoPin, HIGH);
                    afstandUltrasone = tijdUltrasone * 0.034 / 2;
                
                    // 90° links
                    digitalWrite(ain1, HIGH);
                
                    if (afstandUltrasone > 7) {
                      analogWrite(pwma, draai90);
                      delay(1000);
                      digitalWrite(ain1, LOW);
                      analogWrite(pwma, 0);
                    }
                    else {
                      // 180° draaien
                      digitalWrite(ain1, HIGH);
                      analogWrite(pwma, draai90);
                      delay(tijdvoorachter * 2);
                      digitalWrite(ain1, LOW);
                      analogWrite(pwma, 0);
                
                      // Doorgaan
                      digitalWrite(ain1, HIGH);
                      analogWrite(pwma, draai90);
                      delay(1000);
                      digitalWrite(ain1, LOW);
                      analogWrite(pwma, 0);
                    }
                
                    delay(1000);
                
                    // Ultrasone check
                    digitalWrite(trigPin, LOW);
                    delayMicroseconds(2);
                    digitalWrite(trigPin, HIGH);
                    delayMicroseconds(10);
                    digitalWrite(trigPin, LOW);
                
                    tijdUltrasone = pulseIn(echoPin, HIGH);
                    afstandUltrasone = tijdUltrasone * 0.034 / 2;
                
                    // 3s achteruit
                    digitalWrite(ain2, HIGH);
                    digitalWrite(bin2, HIGH);
                
                    if (afstandUltrasone > 7) { 
                      analogWrite(pwma, 255);
                      analogWrite(pwmb, 255);
                      delay(tijdvoorachter);
                      digitalWrite(ain2, LOW);
                      digitalWrite(bin2, LOW);
                    }
                    else {
                      // 180° draaien
                      digitalWrite(ain1, HIGH);
                      analogWrite(pwma, draai90);
                      delay(tijdvoorachter * 2);
                      digitalWrite(ain1, LOW);
                      analogWrite(pwma, 0);
                
                      // Doorgaan
                      analogWrite(pwma, 255);
                      analogWrite(pwmb, 255);
                      delay(tijdvoorachter);
                      digitalWrite(ain2, LOW);
                      digitalWrite(bin2, LOW);
                    }
                
                    delay(1000);
                
                    // 90° rechts
                    digitalWrite(bin1, HIGH);
                    analogWrite(pwmb, draai90);
                    delay(1000);
                    digitalWrite(bin1, LOW);
                    analogWrite(pwma, 0);
                
                    delay(1000);
                  }
                }
